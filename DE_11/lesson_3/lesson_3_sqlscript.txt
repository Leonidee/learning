START TRANSACTION;

CREATE TABLE customer
(
    c_custkey    integer,
    c_name       VARCHAR(25),
    c_address    VARCHAR(40),
    c_nationkey  integer,
    c_phone      CHAR(15),
    c_acctbal    NUMERIC(15, 2),
    c_mktsegment CHAR(10),
    c_comment    VARCHAR(117),
    n_emp        CHAR(2)
);

CREATE TABLE nation
(
    n_nationkey integer,
    n_name      CHAR(25),
    n_regionkey INTEGER,
    n_comment   VARCHAR(152),
    n_emp       CHAR(2)
);

CREATE TABLE region
(
    r_regionkey integer,
    r_name      CHAR(25),
    r_comment   VARCHAR(152),
    n_emp       CHAR(2)
);


CREATE TABLE part
(
    p_partkey     integer,
    p_name        VARCHAR(55),
    p_mfgr        CHAR(25),
    p_brand       CHAR(10),
    p_type        VARCHAR(25),
    p_size        integer,
    p_container   CHAR(10),
    p_retailprice numeric(15, 2),
    p_comment     VARCHAR(23),
    n_emp         CHAR(2)
);

CREATE TABLE partsupp
(
    ps_partkey    integer,
    ps_suppkey    integer,
    ps_availqty   integer,
    ps_supplycost numeric(15, 2),
    ps_comment    VARCHAR(199),
    n_emp         CHAR(2)
);

CREATE TABLE supplier
(
    s_suppkey   integer,
    s_name      CHAR(25),
    s_address   VARCHAR(40),
    s_nationkey INTEGER,
    s_phone     CHAR(15),
    s_acctbal   NUMERIC(15, 2),
    s_comment   VARCHAR(101),
    n_emp       CHAR(2)
);

COMMIT;
